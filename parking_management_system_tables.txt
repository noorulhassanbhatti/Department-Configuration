CREATE TABLE  "CUSTOMER_TABLE" 
   (	"CUSTOMER_ID" VARCHAR2(20) NOT NULL ENABLE, 
	"CUSTOMER_NAME" VARCHAR2(50) NOT NULL ENABLE, 
	"CUSTOMER_EMAIL" VARCHAR2(50) NOT NULL ENABLE, 
	"PHN_NBR" VARCHAR2(13) NOT NULL ENABLE, 
	"EMG_PHN_NBR" VARCHAR2(13) NOT NULL ENABLE, 
	 CONSTRAINT "CUSTOMER_TABLE_PK" PRIMARY KEY ("CUSTOMER_ID") ENABLE, 
	 CONSTRAINT "CHK_ID" CHECK (customer_id>0) ENABLE
   )

CREATE TABLE  "LOCATION_TABLE" 
   (	"LOCATION_ID" VARCHAR2(20) NOT NULL ENABLE, 
	"BLOCK_NUM" VARCHAR2(20) NOT NULL ENABLE, 
	"ROW_NUM" VARCHAR2(20) NOT NULL ENABLE, 
	"POSITION" VARCHAR2(20) NOT NULL ENABLE, 
	 CONSTRAINT "LOCATION_TABLE_PK" PRIMARY KEY ("LOCATION_ID") ENABLE, 
	 CONSTRAINT "CHK_BLOCK" CHECK (BLOCK_NUM BETWEEN 1 AND 5) ENABLE, 
	 CONSTRAINT "CHK_POSITION" CHECK (POSITION BETWEEN 1 AND 30) ENABLE, 
	 CONSTRAINT "CHK_ROW_NUM" CHECK (ROW_NUM BETWEEN 1 AND 30) ENABLE, 
	 CONSTRAINT "LOCATION_TABLE_FK1" FOREIGN KEY ("LOCATION_ID")
	  REFERENCES  "CUSTOMER_TABLE" ("CUSTOMER_ID") ENABLE
   )

CREATE TABLE  "VEHICLE_TABLE" 
   (	"VEHI_OWNER_ID" VARCHAR2(20) NOT NULL ENABLE, 
	"VEHI_MODEL" VARCHAR2(20) NOT NULL ENABLE, 
	"VEHI_TYPE" VARCHAR2(20) NOT NULL ENABLE, 
	"VEHI_COLOR" VARCHAR2(20) NOT NULL ENABLE, 
	"VEHI_NUMBER" VARCHAR2(20) NOT NULL ENABLE, 
	"FEE" NUMBER(4,0) NOT NULL ENABLE, 
	 CONSTRAINT "VEHICAL_TABLE_PK" PRIMARY KEY ("VEHI_OWNER_ID") ENABLE, 
	 CONSTRAINT "CHK_VEHICLE_MODEL" CHECK (VEHI_MODEL BETWEEN 2000 AND 2023)
 ENABLE, 
	 CONSTRAINT "VEHICAL_TABLE_FK1" FOREIGN KEY ("VEHI_OWNER_ID")
	  REFERENCES  "CUSTOMER_TABLE" ("CUSTOMER_ID") ENABLE
   )
